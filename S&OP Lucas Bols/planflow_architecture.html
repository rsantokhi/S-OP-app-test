Waarom ECS Fargate en niet EC2 direct?
Fargate draait Docker containers zonder dat je servers beheert. Je betaalt per seconde gebruik, schaalt automatisch bij piekbelasting, en de blue-green deployment werkt zonder manuele SSH sessies. Voor PlanFlow — met wisselende gebruikspieken tijdens S&OP cycli — is dit de goedkoopste en meest betrouwbare keuze.
02
Database Migraties met Flyway
Elke DB-wijziging is een genummerd bestand — nooit handmatig SQL op productie
Het principe: forward-only, altijd backward compatible
Flyway houdt bij welke migraties al gedraaid zijn. Elke nieuwe versie van PlanFlow brengt zijn eigen migratiescripts mee. Nieuwe kolommen worden altijd nullable toegevoegd — de oude app-versie breekt niet als de nieuwe DB al live staat. Dit is het fundament van zero-downtime deploys.
Migratiestructuur
db/migrations/

# Flyway commando op productie:
flyway migrate -url=jdbc:oracle:thin:@host:1521/PLANFLOW
Gouden regels voor migraties
✓Altijd forward-only — nooit een bestaande migratie aanpassen
✓Nieuwe kolommen altijd NULL-able — oude app blijft werken
✓Grote data-migraties in aparte R__ (repeatable) scripts
✓Migratie draait VOOR nieuwe app-versie goes live
△Nooit een kolom droppen in dezelfde sprint als de app-update
△Rename = nieuw kolom + data kopiëren + oud later droppen

-- V1__init_planflow_schema.sql
-- ============================================================
-- PlanFlow initial schema — Oracle DB 19c
-- Flyway migration V1 — runs once on first deploy
-- ============================================================

-- Users (synced from Azure AD)
CREATE TABLE PLANFLOW.USERS (
    USER_ID        VARCHAR2(100)  PRIMARY KEY,
    EMAIL          VARCHAR2(200)  NOT NULL UNIQUE,
    DISPLAY_NAME   VARCHAR2(200),
    AD_GROUPS      VARCHAR2(1000),
    LAST_LOGIN     TIMESTAMP,
    IS_ACTIVE      NUMBER(1)      DEFAULT 1
);

-- SKU master (synced from Oracle Fusion)
CREATE TABLE PLANFLOW.ITEMS (
    ITEM_CODE      VARCHAR2(50)   PRIMARY KEY,
    DESCRIPTION    VARCHAR2(500),
    CATEGORY       VARCHAR2(100),
    SUPPLIER_CODE  VARCHAR2(50),
    LEAD_TIME_DAYS NUMBER(3),
    MIN_LOT        NUMBER(6),
    INV_VALUE_UNIT NUMBER(12,4),
    LAST_SYNC      TIMESTAMP      DEFAULT SYSTIMESTAMP,
    IS_ACTIVE      NUMBER(1)      DEFAULT 1
);

-- Demand forecast (versioned per SKU per period)
CREATE TABLE PLANFLOW.DEMAND_FORECAST (
    FORECAST_ID    NUMBER         GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    ITEM_CODE      VARCHAR2(50)   NOT NULL REFERENCES PLANFLOW.ITEMS(ITEM_CODE),
    LOCATION       VARCHAR2(100),
    PERIOD_YEAR    NUMBER(4)      NOT NULL,
    PERIOD_MONTH   NUMBER(2)      NOT NULL,
    STAT_FORECAST  NUMBER(12,2),
    FINAL_FORECAST NUMBER(12,2),
    VERSION        NUMBER(3)      DEFAULT 1,
    STATUS         VARCHAR2(20)   DEFAULT 'DRAFT',
    CREATED_BY     VARCHAR2(100)  REFERENCES PLANFLOW.USERS(USER_ID),
    CREATED_AT     TIMESTAMP      DEFAULT SYSTIMESTAMP,
    APPROVED_BY    VARCHAR2(100)  REFERENCES PLANFLOW.USERS(USER_ID),
    APPROVED_AT    TIMESTAMP,
    CONSTRAINT chk_fc_status CHECK (STATUS IN ('DRAFT','SUBMITTED','APPROVED'))
);

-- Purchase orders (workflow)
CREATE TABLE PLANFLOW.PURCHASE_ORDERS (
    PO_ID          NUMBER         GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    ORACLE_PO_ID   VARCHAR2(50),
    ITEM_CODE      VARCHAR2(50)   NOT NULL REFERENCES PLANFLOW.ITEMS(ITEM_CODE),
    SUPPLIER_CODE  VARCHAR2(50)   NOT NULL,
    QTY_ORDERED    NUMBER(12,2)   NOT NULL,
    UNIT_PRICE     NUMBER(12,4),
    NEED_BY_DATE   DATE           NOT NULL,
    STATUS         VARCHAR2(20)   DEFAULT 'DRAFT',
    CREATED_BY     VARCHAR2(100)  REFERENCES PLANFLOW.USERS(USER_ID),
    CREATED_AT     TIMESTAMP      DEFAULT SYSTIMESTAMP,
    APPROVED_BY    VARCHAR2(100)  REFERENCES PLANFLOW.USERS(USER_ID),
    APPROVED_AT    TIMESTAMP,
    REJECTION_NOTE VARCHAR2(1000),
    CONSTRAINT chk_po_status CHECK (STATUS IN ('DRAFT','PENDING','APPROVED','SENT','REJECTED'))
);

-- Audit log (INSERT ONLY — no UPDATE/DELETE permissions for app users)
CREATE TABLE PLANFLOW.AUDIT_LOG (
    LOG_ID         NUMBER         GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    USER_ID        VARCHAR2(100),
    ACTION         VARCHAR2(100),
    TABLE_NAME     VARCHAR2(100),
    RECORD_ID      VARCHAR2(100),
    OLD_VALUES     CLOB,
    NEW_VALUES     CLOB,
    IP_ADDRESS     VARCHAR2(50),
    SESSION_ID     VARCHAR2(200),
    LOG_TIME       TIMESTAMP      DEFAULT SYSTIMESTAMP
);

-- Feature flags (per AD group)
CREATE TABLE PLANFLOW.FEATURE_FLAGS (
    FLAG_NAME      VARCHAR2(100)  PRIMARY KEY,
    AD_GROUP       VARCHAR2(100)  DEFAULT 'ALL',
    IS_ENABLED     NUMBER(1)      DEFAULT 0,
    DESCRIPTION    VARCHAR2(500),
    UPDATED_AT     TIMESTAMP      DEFAULT SYSTIMESTAMP
);

-- Indexes
CREATE INDEX idx_fc_item_period ON PLANFLOW.DEMAND_FORECAST(ITEM_CODE, PERIOD_YEAR, PERIOD_MONTH);
CREATE INDEX idx_po_status ON PLANFLOW.PURCHASE_ORDERS(STATUS);
CREATE INDEX idx_audit_time ON PLANFLOW.AUDIT_LOG(LOG_TIME);
CREATE INDEX idx_audit_user ON PLANFLOW.AUDIT_LOG(USER_ID);

Blue-Green Deployment
Nieuwe versie live in 3 minuten — rollback in 60 seconden
1. Test
npm test + lint
→
2. Build
Docker image + ECR
→
3. Flyway
DB migraties draaien
→
4. Deploy GREEN
Nieuwe versie standby
→
5. Health check
/health endpoint
→
6. Cutover
ALB → GREEN live
→
7. BLUE standby
60s rollback mogelijk
Wat gebruikers meemaken tijdens een update
✓Geen pagina die offline gaat — de ALB stuurt verkeer naadloos door
✓Actieve sessies blijven bestaan — tokens zijn stateless (JWT)
✓Forecasts die iemand aan het bewerken is gaan niet verloren — opgeslagen in DB
✓Maximaal 1-2 requests die een 502 zien bij de daadwerkelijke cutover
Als een update misgaat — rollback procedure
1Health check GREEN faalt → pipeline stopt automatisch
2BLUE blijft live — gebruikers merken niets
3Handmatige rollback: ALB regel terugzetten naar BLUE (60 sec)
4DB migraties zijn al gedraaid — backward compatible dus geen probleem
△Dát is waarom elke migratie backward compatible moet zijn

CI/CD Pipeline — GitHub Actions
Automatisch testen, bouwen en deployen bij elke push naar main
.github/workflows/deploy.yml GitHub Actions YAML 
# .github/workflows/deploy.yml
# ============================================================
# PlanFlow CI/CD Pipeline — GitHub Actions
# Blue-green deployment naar AWS
# ============================================================

name: PlanFlow Deploy

on:
  push:
    branches: [main]
  workflow_dispatch:
    inputs:
      environment:
        description: 'Target environment'
        required: true
        default: 'staging'
        type: choice
        options: [staging, production]

env:
  AWS_REGION: eu-west-1
  ECR_REPOSITORY: planflow-app
  APP_NAME: planflow

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      - run: npm ci
      - run: npm test
      - run: npm run lint

  build-and-push:
    needs: test
    runs-on: ubuntu-latest
    outputs:
      image-tag: ${{ steps.meta.outputs.tags }}
    steps:
      - uses: actions/checkout@v4

      - name: Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to Amazon ECR
        id: login-ecr
        uses: aws-actions/amazon-ecr-login@v2

      - name: Build, tag, push Docker image
        id: meta
        env:
          ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
          IMAGE_TAG: ${{ github.sha }}
        run: |
          docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG .
          docker push $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG
          echo "tags=$ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG" >> $GITHUB_OUTPUT

  migrate-db:
    needs: build-and-push
    runs-on: ubuntu-latest
    environment: ${{ github.event.inputs.environment || 'staging' }}
    steps:
      - uses: actions/checkout@v4

      - name: Run Flyway migrations
        # Flyway runs BEFORE new app code goes live
        # Safe: migrations are always backward compatible
        run: |
          docker run --rm             -e FLYWAY_URL="${{ secrets.DB_JDBC_URL }}"             -e FLYWAY_USER="${{ secrets.DB_USER }}"             -e FLYWAY_PASSWORD="${{ secrets.DB_PASSWORD }}"             -v $(pwd)/db/migrations:/flyway/sql             flyway/flyway:10 migrate
        # If migration fails → pipeline stops, old app keeps running, NO downtime

  deploy-green:
    needs: migrate-db
    runs-on: ubuntu-latest
    environment: ${{ github.event.inputs.environment || 'staging' }}
    steps:
      - name: Deploy to GREEN environment
        # Deploy new version to green — blue still serves traffic
        uses: aws-actions/amazon-ecs-deploy-task-definition@v1
        with:
          task-definition: planflow-task-def.json
          service: planflow-green
          cluster: planflow-cluster
          wait-for-service-stability: true

      - name: Health check GREEN
        run: |
          sleep 15
          STATUS=$(curl -s -o /dev/null -w "%{http_code}"             https://green.planflow-internal.com/health)
          if [ "$STATUS" != "200" ]; then
            echo "Green health check failed: $STATUS"
            exit 1
          fi
          echo "Green is healthy — switching traffic"

      - name: Switch traffic BLUE → GREEN
        # Instant cutover — users see new version within 30 seconds
        run: |
          aws elbv2 modify-rule             --rule-arn ${{ secrets.ALB_RULE_ARN }}             --actions '[{"Type":"forward","TargetGroupArn":"${{ secrets.GREEN_TG_ARN }}"}]'

      - name: Notify team
        run: |
          echo "Deploy ${{ github.sha }} live on ${{ github.event.inputs.environment || 'staging' }}"
          # Add Slack/Teams notification here

src/server.js Node.js — productie backend 
// src/server.js
// ============================================================
// PlanFlow Backend — productie entry point
// ============================================================
const express    = require('express');
const helmet     = require('helmet');
const rateLimit  = require('express-rate-limit');
const passport   = require('passport');
const { BearerStrategy } = require('passport-azure-ad');
const oracledb   = require('oracledb');
const winston    = require('winston');

const app = express();

// ── LOGGER ──────────────────────────────────────────────────
const log = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.json()
  ),
  transports: [new winston.transports.Console()]
});

// ── SECURITY MIDDLEWARE ──────────────────────────────────────
app.use(helmet());
app.use(express.json({ limit: '1mb' }));
app.use(rateLimit({ windowMs: 15*60*1000, max: 500 }));

// ── HEALTH CHECK (voor AWS ALB + Flyway pipeline) ────────────
app.get('/health', async (req, res) => {
  try {
    const conn = await oracledb.getConnection();
    await conn.execute('SELECT 1 FROM DUAL');
    await conn.close();
    res.json({ status: 'ok', db: 'connected', ts: new Date() });
  } catch (err) {
    res.status(503).json({ status: 'error', db: 'disconnected' });
  }
});

// ── AZURE AD SSO ─────────────────────────────────────────────
passport.use(new BearerStrategy({
  identityMetadata: `https://login.microsoftonline.com/${process.env.AZURE_TENANT_ID}/v2.0/.well-known/openid-configuration`,
  clientID: process.env.AZURE_CLIENT_ID,
  validateIssuer: true,
  loggingLevel: 'warn'
}, async (token, done) => {
  try {
    const conn = await oracledb.getConnection();
    const result = await conn.execute(
      'SELECT * FROM PLANFLOW.USERS WHERE USER_ID = :id AND IS_ACTIVE = 1',
      { id: token.oid }
    );
    await conn.close();
    if (!result.rows.length) return done(null, false);
    done(null, { ...result.rows[0], adGroups: token.groups || [] });
  } catch (err) {
    done(err);
  }
}));

// ── RBAC MIDDLEWARE ──────────────────────────────────────────
const requireRole = (role) => (req, res, next) => {
  if (!req.user?.adGroups?.includes(role)) {
    return res.status(403).json({ error: 'Onvoldoende rechten' });
  }
  next();
};

// ── ORACLE DB POOL ───────────────────────────────────────────
async function initDB() {
  await oracledb.createPool({
    user:          process.env.DB_USER,
    password:      process.env.DB_PASSWORD,
    connectString: process.env.DB_CONNECT_STRING,
    poolMin: Number(process.env.DB_POOL_MIN) || 4,
    poolMax: Number(process.env.DB_POOL_MAX) || 20,
    poolIncrement: 2
  });
  log.info('Oracle DB pool initialized');
}

// ── API ROUTES ────────────────────────────────────────────────
const auth = passport.authenticate('oauth-bearer', { session: false });

// Forecasts ophalen
app.get('/api/forecasts', auth, async (req, res) => {
  const { year, month, category } = req.query;
  const conn = await oracledb.getConnection();
  try {
    const sql = `
      SELECT f.*, i.DESCRIPTION, i.CATEGORY, i.SUPPLIER_CODE
      FROM PLANFLOW.DEMAND_FORECAST f
      JOIN PLANFLOW.ITEMS i ON f.ITEM_CODE = i.ITEM_CODE
      WHERE f.PERIOD_YEAR = :year
        AND f.PERIOD_MONTH = :month
        AND (:cat IS NULL OR i.CATEGORY = :cat)
      ORDER BY i.CATEGORY, i.ITEM_CODE`;
    const result = await conn.execute(sql,
      { year: Number(year), month: Number(month), cat: category || null },
      { outFormat: oracledb.OUT_FORMAT_OBJECT }
    );
    res.json(result.rows);
  } finally {
    await conn.close();
  }
});

// Forecast opslaan + audit log
app.put('/api/forecasts/:id', auth, async (req, res) => {
  const { finalForecast } = req.body;
  const conn = await oracledb.getConnection();
  try {
    // Haal oude waarde op voor audit
    const old = await conn.execute(
      'SELECT FINAL_FORECAST FROM PLANFLOW.DEMAND_FORECAST WHERE FORECAST_ID = :id',
      { id: req.params.id },
      { outFormat: oracledb.OUT_FORMAT_OBJECT }
    );
    // Update
    await conn.execute(
      'UPDATE PLANFLOW.DEMAND_FORECAST SET FINAL_FORECAST=:fc, STATUS=:st WHERE FORECAST_ID=:id',
      { fc: finalForecast, st: 'DRAFT', id: req.params.id }
    );
    // Audit log
    await conn.execute(`
      INSERT INTO PLANFLOW.AUDIT_LOG
        (USER_ID, ACTION, TABLE_NAME, RECORD_ID, OLD_VALUES, NEW_VALUES, IP_ADDRESS)
      VALUES (:uid, 'FORECAST_UPDATE', 'DEMAND_FORECAST', :rid, :old, :new, :ip)`,
      {
        uid: req.user.USER_ID,
        rid: req.params.id,
        old: JSON.stringify(old.rows[0]),
        new: JSON.stringify({ FINAL_FORECAST: finalForecast }),
        ip:  req.ip
      }
    );
    await conn.commit();
    res.json({ ok: true });
  } catch (err) {
    await conn.rollback();
    throw err;
  } finally {
    await conn.close();
  }
});

// Forecast goedkeuren (PLANFLOW_APPROVER vereist)
app.post('/api/forecasts/:id/approve', auth,
  requireRole('PLANFLOW_APPROVER'),
  async (req, res) => {
    const conn = await oracledb.getConnection();
    try {
      await conn.execute(`
        UPDATE PLANFLOW.DEMAND_FORECAST
        SET STATUS='APPROVED', APPROVED_BY=:uid, APPROVED_AT=SYSTIMESTAMP
        WHERE FORECAST_ID=:id`,
        { uid: req.user.USER_ID, id: req.params.id }
      );
      await conn.execute(`
        INSERT INTO PLANFLOW.AUDIT_LOG (USER_ID,ACTION,TABLE_NAME,RECORD_ID,IP_ADDRESS)
        VALUES (:uid,'FORECAST_APPROVE','DEMAND_FORECAST',:rid,:ip)`,
        { uid: req.user.USER_ID, rid: req.params.id, ip: req.ip }
      );
      await conn.commit();
      res.json({ ok: true });
    } finally {
      await conn.close();
    }
  }
);

// PO aanmaken als Draft in Oracle Fusion
app.post('/api/purchase-orders', auth,
  requireRole('PLANFLOW_APPROVER'),
  async (req, res) => {
    const { itemCode, qty, supplierId, needByDate } = req.body;
    // Stap 1: opslaan in lokale DB
    const conn = await oracledb.getConnection();
    let poId;
    try {
      const result = await conn.execute(`
        INSERT INTO PLANFLOW.PURCHASE_ORDERS
          (ITEM_CODE, SUPPLIER_CODE, QTY_ORDERED, NEED_BY_DATE, STATUS, CREATED_BY)
        VALUES (:item, :sup, :qty, TO_DATE(:nbd,'YYYY-MM-DD'), 'PENDING', :uid)
        RETURNING PO_ID INTO :poId`,
        { item:itemCode, sup:supplierId, qty, nbd:needByDate, uid:req.user.USER_ID, poId:{dir:oracledb.BIND_OUT,type:oracledb.NUMBER} }
      );
      poId = result.outBinds.poId[0];
      await conn.commit();
    } finally {
      await conn.close();
    }
    // Stap 2: stuur naar Oracle Fusion als Draft PO
    // (zie fusion-client.js — altijd DRAFT, nooit definitief)
    res.json({ ok: true, poId });
  }
);

// ── START ─────────────────────────────────────────────────────
async function start() {
  await initDB();
  app.listen(process.env.PORT || 3000, () => {
    log.info(`PlanFlow running on port ${process.env.PORT || 3000}`);
  });
}
start().catch(err => { log.error(err); process.exit(1); });
